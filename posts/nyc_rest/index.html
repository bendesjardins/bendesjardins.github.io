#Q1a) What are the mean, standard deviation, first quartile, median, third quartile, and maximum of SCORE for each GRADE of restaurants?

scorePerGrade <- restaurant %>% 
  group_by(restaurant$GRADE) %>%
  summarize(
    score_mean = mean(SCORE),
    score_sd = sd(SCORE),
    score_min = min(SCORE),
    score_q1 = quantile(SCORE, probs = .25),
    score_median = median(SCORE),
    score_q1 = quantile(SCORE, probs = .75),
    score_max = max(SCORE) )

#Q1c)

Provide both (1) ggplot code and (2) a simple comment to describe how the distribution of SCORE varies by GRADE and CRITICAL FLAG.

ggplot( data = restaurant) +
  geom_histogram(aes(x = SCORE, fill = restaurant$`CRITICAL FLAG`), bins = 50) +
  facet_grid(. ~ GRADE)

Critical makes up the majority of each grade, and those with the grade A has the highest and most varied scores, those with grade C have much larger scores

#Q1d) Provide both (1) ggplot code and (2) a simple comment to describe how the proportion of CRITICAL FLAG varies by GRADE and BORO.

ggplot(data = restaurant) + 
  geom_bar(aes(x = GRADE, fill = BORO)) +
  facet_wrap(. ~ `CRITICAL FLAG`)

Staten Island and Bronx make up the smallest proportions of boroughs.
Manhattan makes up the largest proportion of each grade. Critical and Not Critical have very similar distributions.

#Q1f) Find the 3 most common names of restaurants (DBA) in each BORO. If the third most common DBA values are multiple, please include all the DBA values.

q2f <- restaurant %>% 
  select(DBA, BORO) %>% 
  group_by(BORO, DBA) %>% 
  summarize(n = n()) %>% 
  mutate(ranking = dense_rank(-n)) %>% 
  filter(ranking <= 3) %>% 
  arrange(BORO, ranking)
`summarise()` has grouped output by 'BORO'. You can override using the
`.groups` argument.
q2f_ <- restaurant %>% 
  group_by(DBA) %>% 
  count() %>% 
  arrange(-n)

#Q1g) For all the DBA values that appear in the result of Q1f, find the DBA value that is most likely to commit critical violation.

q2g <- restaurant %>% 
  filter(DBA %in% q2f$DBA) %>% 
  group_by(DBA, `CRITICAL FLAG`) %>% 
  count() %>% 
  group_by(DBA) %>% 
  mutate(lag_n = lag(n),
         tot = sum(n),
         prop_crit = lag_n / tot) %>% 
  select(DBA, prop_crit) %>% 
  filter(!is.na(prop_crit)) %>% 
  arrange(-prop_crit)
